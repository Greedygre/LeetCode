func diStringMatch(S string) []int {
	len := len(S)
	result := []int{}
	small := 0
	big := len
	//result = append(result, 0)

	for i := 0; i < len; i++ {
		if S[i] == 'I' {

			result = append(result, small)
			small++
		} else {

			result = append(result, big)
			big--
		}
	}
	result = append(result, big)
	return result
}
